#!/usr/bin/env bash # # expand - Ubuntu Filesystem Expansion Tool (LVM and plain partitions) # https://github.com/SolubleSpork/expand # # Usage: curl -sL solublespork.github.io/expand | sudo bash # set -euo pipefail VERSION="2.0.0" DRY_RUN=false AUTO_MODE=false HAVE_LVM=false # Colors RED='\033[0;31m' GREEN='\033[0;32m' YELLOW='\033[1;33m' BLUE='\033[0;34m' CYAN='\033[0;36m' DIM='\033[2m' BOLD='\033[1m' NC='\033[0m' # Global array to store scan results declare -a EXPANDABLE_ITEMS=() print_success() { echo -e "${GREEN}✓${NC} $1"; } print_error() { echo -e "${RED}✗${NC} $1"; } print_warning() { echo -e "${YELLOW}!${NC} $1"; } confirm() { local response echo -en "${YELLOW}?${NC} $1 [Y/n] " read -r response < /dev/tty [[ ! "$response" =~ ^[Nn]$ ]] } run_cmd() { local cmd="$1" if $DRY_RUN; then echo -e " ${DIM}[dry-run]${NC} $cmd" else echo -e " ${DIM}Running:${NC} $cmd" eval "$cmd" fi } check_root() { if [[ $EUID -ne 0 ]]; then print_error "This script must be run as root (use sudo)" exit 1 fi } check_lvm() { if command -v pvs &> /dev/null; then HAVE_LVM=true else HAVE_LVM=false fi } check_growpart() { if ! command -v growpart &> /dev/null; then print_error "growpart not found. Install with: apt install cloud-guest-utils" exit 1 fi } is_supported_fstype() { case "$1" in ext4|ext3|ext2|xfs|btrfs) return 0 ;; *) return 1 ;; esac } # Disks already rescanned this run, so a disk with multiple partitions # (e.g. sda2 + sda3 both on sda) only gets rescanned once. declare -a RESCANNED_DISKS=() rescan_disk_once() { local disk_base="$1" [[ " ${RESCANNED_DISKS[*]} " == *" $disk_base "* ]] && return # Best-effort: not all controllers expose this (e.g. virtio-blk); harmless no-op if missing. { echo 1 > "/sys/class/block/$disk_base/device/rescan"; } 2>/dev/null || true RESCANNED_DISKS+=("$disk_base") } # Grow the partition backing $1 (a partition device, e.g. /dev/sda3) to match # the real disk size. Non-destructive: only rewrites the partition table. grow_partition() { local part_device="$1" part_base disk_base disk partnum growpart_out part_base=$(basename "$part_device") disk_base=$(lsblk -no PKNAME "$part_device" 2>/dev/null | head -1) || true [[ -z "$disk_base" ]] && return disk="/dev/$disk_base" rescan_disk_once "$disk_base" [[ -f "/sys/class/block/$part_base/partition" ]] || return partnum=$(cat "/sys/class/block/$part_base/partition" 2>/dev/null || echo "") [[ -z "$partnum" ]] && return growpart_out=$(growpart "$disk" "$partnum" 2>&1) && { print_success "Grew partition $part_device to match disk size" } || { [[ "$growpart_out" != *NOCHANGE* ]] && print_warning "growpart $disk $partnum: $growpart_out" } } # Grow partitions (and PVs, where LVM is in use) to match the real (possibly # just-resized-in-Proxmox) disk size, so the free-space scan below sees # accurate numbers whether or not LVM is involved. Non-destructive to # filesystem data: only touches the partition table and PV metadata. Always # runs for real, even in dry-run mode, so scan results stay accurate; # lvextend/filesystem growth remain gated behind dry-run/confirm. sync_storage_layers() { local pv_device pv_list=" " if $HAVE_LVM; then while read -r pv_device; do pv_device=$(echo "$pv_device" | xargs) [[ -z "$pv_device" ]] && continue pv_list+="$pv_device " grow_partition "$pv_device" local before after before=$(pvs --noheadings -o pv_size --units b "$pv_device" 2>/dev/null | xargs) || true pvresize "$pv_device" &> /dev/null || true after=$(pvs --noheadings -o pv_size --units b "$pv_device" 2>/dev/null | xargs) || true [[ -n "$before" && -n "$after" && "$before" != "$after" ]] && print_success "Resized PV $pv_device" done < <(pvs --noheadings -o pv_name 2>/dev/null) fi # Plain (non-LVM) mounted partitions: grow directly, no PV involved. local name fstype mountpoint while read -r name fstype mountpoint; do [[ -z "$name" || -z "$mountpoint" ]] && continue is_supported_fstype "$fstype" || continue [[ "$pv_list" == *" /dev/$name "* ]] && continue grow_partition "/dev/$name" done < <(lsblk -rno NAME,FSTYPE,MOUNTPOINT 2>/dev/null) } # Convert size string to bytes for comparison size_to_bytes() { local size="$1" local num unit num=$(echo "$size" | sed 's/[^0-9.]//g') unit=$(echo "$size" | sed 's/[0-9.<]//g' | tr '[:lower:]' '[:upper:]') case "$unit" in T|TB|TIB) echo "$num * 1024 * 1024 * 1024 * 1024" | bc 2>/dev/null || echo 0 ;; G|GB|GIB) echo "$num * 1024 * 1024 * 1024" | bc 2>/dev/null || echo 0 ;; M|MB|MIB) echo "$num * 1024 * 1024" | bc 2>/dev/null || echo 0 ;; K|KB|KIB) echo "$num * 1024" | bc 2>/dev/null || echo 0 ;; *) echo "$num" ;; esac } # Main scanning function - finds all expansion opportunities scan_system() { EXPANDABLE_ITEMS=() local found_any=false local pv_list=" " if $HAVE_LVM; then # Get VG info (free space and total size) declare -A vg_free declare -A vg_size while read -r vg_name free_space total_size; do vg_name=$(echo "$vg_name" | xargs) free_space=$(echo "$free_space" | xargs) total_size=$(echo "$total_size" | xargs) if [[ -n "$vg_name" ]]; then vg_free["$vg_name"]="$free_space" vg_size["$vg_name"]="$total_size" fi done < <(vgs --noheadings -o vg_name,vg_free,vg_size --units g 2>/dev/null) # Scan each LV while read -r lv_line; do [[ -z "$lv_line" ]] && continue local lv_path lv_size vg_name lv_path=$(echo "$lv_line" | awk '{print $1}' | xargs) lv_size=$(echo "$lv_line" | awk '{print $2}' | xargs) vg_name=$(echo "$lv_line" | awk '{print $3}' | xargs) [[ -z "$lv_path" ]] && continue local mountpoint fstype fs_size fs_used fs_pct mountpoint=$(findmnt -n -o TARGET "$lv_path" 2>/dev/null || echo "") fstype=$(blkid -o value -s TYPE "$lv_path" 2>/dev/null || echo "unknown") # Get filesystem usage if mounted if [[ -n "$mountpoint" ]]; then read -r fs_size fs_used fs_pct < <(df -h "$mountpoint" 2>/dev/null | tail -1 | awk '{print $2, $3, $5}') else fs_size="$lv_size" fs_used="-" fs_pct="-" fi # Check if VG has free space local free="${vg_free[$vg_name]:-0}" local free_bytes free_bytes=$(size_to_bytes "$free") # Only show if there's meaningful free space (> 100MB) if (( $(echo "$free_bytes > 104857600" | bc 2>/dev/null || echo 0) )); then found_any=true # Calculate potential size (current + free) local current_num free_num potential_size current_num=$(echo "$fs_size" | sed 's/[^0-9.]//g') free_num=$(echo "$free" | sed 's/[^0-9.]//g') potential_size=$(echo "$current_num + $free_num" | bc 2>/dev/null || echo "?") # format: lv_path|vg_name|mountpoint|fstype|current_fs_size|potential_size|used|used_pct EXPANDABLE_ITEMS+=("$lv_path|$vg_name|$mountpoint|$fstype|$fs_size|${potential_size}G|$fs_used|$fs_pct") fi done < <(lvs --noheadings -o lv_path,lv_size,vg_name --units g 2>/dev/null) while read -r pv_device; do pv_device=$(echo "$pv_device" | xargs) [[ -n "$pv_device" ]] && pv_list+="$pv_device " done < <(pvs --noheadings -o pv_name 2>/dev/null) fi # Scan plain (non-LVM) mounted partitions with a supported filesystem local name fstype mountpoint while read -r name fstype mountpoint; do [[ -z "$name" || -z "$mountpoint" ]] && continue is_supported_fstype "$fstype" || continue [[ "$pv_list" == *" /dev/$name "* ]] && continue local part_bytes fs_size fs_used fs_pct fs_bytes part_bytes=$(blockdev --getsize64 "/dev/$name" 2>/dev/null || echo 0) read -r fs_size fs_used fs_pct < <(df -h "$mountpoint" 2>/dev/null | tail -1 | awk '{print $2, $3, $5}') fs_bytes=$(size_to_bytes "$fs_size") # Only show if the partition has meaningfully more room than the filesystem (> 100MB) local slack slack=$(echo "$part_bytes - $fs_bytes" | bc 2>/dev/null || echo 0) if (( $(echo "$slack > 104857600" | bc 2>/dev/null || echo 0) )); then found_any=true local potential_size potential_size=$(echo "scale=1; $part_bytes / 1073741824" | bc 2>/dev/null || echo "?") # vg_name left empty to mark this as a plain (non-LVM) partition EXPANDABLE_ITEMS+=("/dev/$name||$mountpoint|$fstype|$fs_size|${potential_size}G|$fs_used|$fs_pct") fi done < <(lsblk -rno NAME,FSTYPE,MOUNTPOINT 2>/dev/null) $found_any } show_scan_results() { local count=${#EXPANDABLE_ITEMS[@]} if [[ $count -eq 0 ]]; then echo "" echo -e "${DIM}No expansion opportunities found.${NC}" echo "" echo "No free space available to expand into. If you recently expanded" echo "the disk in Proxmox, double-check the resize completed there." exit 0 fi echo "" if [[ $count -eq 1 ]]; then echo -e "${GREEN}Found 1 volume that can be expanded:${NC}" else echo -e "${GREEN}Found $count volumes that can be expanded:${NC}" fi echo "" local i=1 for item in "${EXPANDABLE_ITEMS[@]}"; do IFS='|' read -r lv_path vg_name mountpoint fstype current_size potential_size used used_pct <<< "$item" # Calculate how much can be added local current_num potential_num gain current_num=$(echo "$current_size" | sed 's/[^0-9.]//g') potential_num=$(echo "$potential_size" | sed 's/[^0-9.]//g') gain=$(echo "$potential_num - $current_num" | bc 2>/dev/null || echo "?") # Display name: use mountpoint if available, otherwise LV path local display_name="${mountpoint:-$lv_path}" echo -e " ${BOLD}$i)${NC} ${CYAN}$display_name${NC} ${DIM}($fstype)${NC}" echo -e " ${DIM}$lv_path${NC}" echo -e " Filesystem: ${BOLD}$current_size${NC} → could be ${GREEN}$potential_size${NC} ${GREEN}(+${gain}G)${NC}" echo -e " Currently using: $used ($used_pct)" echo "" ((i++)) done } show_menu() { local count=${#EXPANDABLE_ITEMS[@]} if [[ $count -gt 1 ]]; then echo -e " ${BOLD}a)${NC} Expand all" fi echo -e " ${BOLD}q)${NC} Quit" echo "" # Build prompt based on count local prompt if [[ $count -eq 1 ]]; then prompt="Select [1/q]: " else prompt="Select [1-$count/a/q]: " fi echo -en "$prompt" read -r choice < /dev/tty case "$choice" in q|Q) echo "Exiting." exit 0 ;; a|A) if [[ $count -gt 1 ]]; then expand_all_lvs else print_error "Invalid input" fi ;; [1-9]*) if (( choice >= 1 && choice <= count )); then expand_single_lv "$((choice - 1))" else print_error "Invalid selection" fi ;; *) print_error "Invalid input" ;; esac } expand_single_lv() { local idx="$1" IFS='|' read -r lv_path vg_name mountpoint fstype current_size potential_size used used_pct <<< "${EXPANDABLE_ITEMS[$idx]}" local display_name="${mountpoint:-$lv_path}" echo "" echo -e "${BOLD}Expanding:${NC} ${CYAN}$display_name${NC}" echo -e "${DIM}$lv_path${NC}" echo "" local step=1 echo "Steps:" if [[ -n "$vg_name" ]]; then echo -e " $step. ${DIM}lvextend -l +100%FREE $lv_path${NC}" ((step++)) fi case "$fstype" in ext4|ext3|ext2) echo -e " $step. ${DIM}resize2fs $lv_path${NC}" ;; xfs) echo -e " $step. ${DIM}xfs_growfs $mountpoint${NC}" ;; btrfs) echo -e " $step. ${DIM}btrfs filesystem resize max $mountpoint${NC}" ;; esac echo "" if $DRY_RUN || confirm "Proceed?"; then [[ -n "$vg_name" ]] && run_cmd "lvextend -l +100%FREE $lv_path" case "$fstype" in ext4|ext3|ext2) run_cmd "resize2fs $lv_path" ;; xfs) [[ -n "$mountpoint" ]] && run_cmd "xfs_growfs $mountpoint" ;; btrfs) [[ -n "$mountpoint" ]] && run_cmd "btrfs filesystem resize max $mountpoint" ;; esac echo "" if ! $DRY_RUN; then print_success "Expansion complete!" [[ -n "$mountpoint" ]] && df -h "$mountpoint" | tail -1 fi fi } expand_all_lvs() { echo "" echo -e "${BOLD}Expanding all volumes...${NC}" echo "" if ! $DRY_RUN && ! $AUTO_MODE; then if ! confirm "Expand all ${#EXPANDABLE_ITEMS[@]} volumes?"; then echo "Cancelled." return fi fi # Extend all LVs (plain partitions were already grown by sync_storage_layers) echo -e "${DIM}Step 1: Extending logical volumes...${NC}" for item in "${EXPANDABLE_ITEMS[@]}"; do IFS='|' read -r lv_path vg_name mountpoint fstype _ _ _ _ <<< "$item" [[ -n "$vg_name" ]] && run_cmd "lvextend -l +100%FREE $lv_path" done echo "" # Then resize all filesystems echo -e "${DIM}Step 2: Resizing filesystems...${NC}" for item in "${EXPANDABLE_ITEMS[@]}"; do IFS='|' read -r lv_path vg_name mountpoint fstype _ _ _ _ <<< "$item" case "$fstype" in ext4|ext3|ext2) run_cmd "resize2fs $lv_path" ;; xfs) [[ -n "$mountpoint" ]] && run_cmd "xfs_growfs $mountpoint" ;; btrfs) [[ -n "$mountpoint" ]] && run_cmd "btrfs filesystem resize max $mountpoint" ;; esac done echo "" if $DRY_RUN; then print_success "Dry-run complete. No changes made." else print_success "All expansions complete!" echo "" echo "New sizes:" df -h --type=ext4 --type=xfs --type=btrfs 2>/dev/null | head -5 || df -h | grep -E '^/dev' | head -5 fi } show_usage() { cat << EOF ${BOLD}expand${NC} v${VERSION} - Filesystem Expansion Tool (LVM and plain partitions) ${BOLD}Usage:${NC} curl -sL solublespork.github.io/expand | sudo bash curl -sL solublespork.github.io/expand | sudo bash -s auto curl -sL solublespork.github.io/expand | sudo bash -s dry-run ${BOLD}Options:${NC} auto Expand everything automatically (no prompts) dry-run Show what would happen without making changes help Show this help ${BOLD}What it does:${NC} 1. growpart (+ pvresize if using LVM) - Grow the partition to match the disk 2. lvextend - Extend LV to use free VG space (LVM only) 3. resize2fs/xfs_growfs/btrfs resize - Grow the filesystem EOF } main() { while [[ $# -gt 0 ]]; do case "$1" in auto|--auto|-a) AUTO_MODE=true; shift ;; dry-run|--dry-run|-d) DRY_RUN=true; shift ;; help|--help|-h) show_usage; exit 0 ;; --version|-v) echo "expand v$VERSION"; exit 0 ;; *) print_error "Unknown option: $1"; exit 1 ;; esac done echo "" echo -e "${BOLD}expand${NC} ${DIM}v${VERSION}${NC}" if $DRY_RUN; then echo -e "${YELLOW}(dry-run mode)${NC}" fi check_root check_lvm check_growpart echo -en "\nSyncing partition/LVM layers... " sync_storage_layers echo "done." echo -en "\nScanning... " if scan_system; then echo "done." show_scan_results if $AUTO_MODE; then expand_all_lvs else show_menu fi else echo "done." echo "" echo -e "${DIM}No expansion opportunities found.${NC}" echo "" echo "No free space available to expand into. If you recently expanded" echo "the disk in Proxmox, double-check the resize completed there." echo "" fi } main "$@"
